OPERATORS:
    Arithmetic: +,-,*,/,%
    Comparison: =, !=, >, >=, <, <=
    Logical: ALL, LIKE, ANY, BETWEEN, AND, OR, NOT, IN
    Bitwise: &, |

Search Query:
    >> SELECT age,fullName 
        FROM student 
        WHERE fullName IN ("name1","name2","name3") 
        AND age BETWEEN 32 AND 45
        OR age > 50
        ORDER BY age DESC   // DESC | ASC
        LIMIT 7;
    
Aggregate Functions:
    - COUNT()
    - MAX()
    - MIN()
    - SUM()
    - AVG()
    >> SELECT id,fullName, AVG(age)
        FROM student
        GROUP BY id,fullName
        ORDER BY AVG(age), id 
        LIMIT 4;
